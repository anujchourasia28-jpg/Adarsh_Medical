        let inventory = [];
        let siteSettings = {};

        function init() {
            // Settings Sync
            db.ref('settings').on('value', (snap) => {
                const defaults = {
                    brandName: "ADARSH MEDICAL",
                    brandSub: "Premium Pharmacy",
                    waNumber: "916261736546",
                    instaUrl: "https://www.instagram.com/adarsh_medicals_?igsh=MW0xZmZ0dWF3NDRxcg==&utm_source=qr",

                    hero: "https://images.unsplash.com/photo-1587854692152-cbe660dbbb88?q=80&w=1000"
                };
                siteSettings = snap.val() || defaults;
                applySettings();
            });

            // Inventory Sync
            db.ref('inventory').on('value', (snap) => {
                const data = snap.val();
                inventory = data ? Object.entries(data).map(([id, item]) => ({...item, id})) : [];
                renderProducts();
                renderStaffInventory();
            });
        }

        function applySettings() {
            const { brandName, brandSub, waNumber, instaUrl, mapUrl, logo, hero } = siteSettings;
            
            // Logo & Hero UI
            const logoImg = document.getElementById('storeLogo');
            const defLogo = document.getElementById('defaultLogo');
            const footLogo = document.getElementById('footerLogo');
            const footDef = document.getElementById('footerLogoDefault');
            
            if (logo) {
                logoImg.src = logo; logoImg.classList.remove('hidden'); defLogo.classList.add('hidden');
                footLogo.src = logo; footLogo.classList.remove('hidden'); footDef.classList.add('hidden');
            } else {
                logoImg.classList.add('hidden'); defLogo.classList.remove('hidden');
                footLogo.classList.add('hidden'); footDef.classList.remove('hidden');
            }
            
            if (hero) document.getElementById('heroImg').src = hero;

            // Brand Text
            document.getElementById('navBrandName').innerText = brandName;
            document.getElementById('navBrandSub').innerText = brandSub;
            document.querySelectorAll('.footerBrand').forEach(el => el.innerText = brandName);
            document.getElementById('displayNumber').innerText = `+${waNumber}`;

            // Social & Map Links
            document.getElementById('navInsta').href = instaUrl;
            document.getElementById('navMap').href = mapUrl;
            document.getElementById('footerInsta').href = instaUrl;
            document.getElementById('footerMap').href = mapUrl;
            document.getElementById('footerWa').href = `https://wa.me/${waNumber}`;

            // Form Sync
            document.getElementById('sLogo').value = logo || "";
            document.getElementById('sHero').value = hero || "";
            document.getElementById('sBrandName').value = brandName;
            document.getElementById('sBrandSub').value = brandSub;
            document.getElementById('sWa').value = waNumber;
            document.getElementById('sInsta').value = instaUrl;
            document.getElementById('sMap').value = mapUrl;
        }

        // CART LOGIC
        function toggleCart() { document.getElementById('cartDrawer').classList.toggle('open'); }
        
        function addToCart(id) {
            const p = inventory.find(i => i.id === id);
            const ex = cart.find(i => i.id === id);
            if(ex) ex.qty++; else cart.push({...p, qty:1});
            updateCart();
            if(!document.getElementById('cartDrawer').classList.contains('open')) toggleCart();
        }

        function updateCart() {
            const total = cart.reduce((a,b)=>a+(b.price*b.qty), 0);
            document.getElementById('cartCount').innerText = cart.reduce((a,b)=>a+b.qty, 0);
            document.getElementById('cartCount').classList.toggle('hidden', cart.length === 0);
            document.getElementById('cartTotal').innerText = `Rs.${total}`;
            document.getElementById('cartItems').innerHTML = cart.map(item => `
                <div class="flex items-center gap-4 bg-white p-4 rounded-2xl border border-slate-100">
                    <div class="flex-1">
                        <h4 class="text-xs font-bold text-slate-800">${item.name}</h4>
                        <p class="text-teal-600 font-black text-[10px]">Rs.${item.price}</p>
                    </div>
                    <div class="flex items-center gap-2 bg-slate-50 p-1 rounded-xl">
                        <button onclick="changeQty('${item.id}', -1)" class="w-7 h-7 rounded-lg bg-white border text-xs text-slate-400 hover:text-red-500 transition-colors">-</button>
                        <span class="text-xs font-bold w-4 text-center">${item.qty}</span>
                        <button onclick="changeQty('${item.id}', 1)" class="w-7 h-7 rounded-lg bg-white border text-xs text-slate-400 hover:text-teal-500 transition-colors">+</button>
                    </div>
                </div>
            `).join('');
        }

        function changeQty(id, d) {
            const i = cart.find(x => x.id === id);
            i.qty += d;
            if(i.qty <= 0) cart = cart.filter(x => x.id !== id);
            updateCart();
        }

        function checkout() {
            const name = document.getElementById('custName').value.trim();
            const phone = document.getElementById('custPhone').value.trim();
            const addr = document.getElementById('custAddr').value.trim();
            const presc = document.getElementById('prescCheck').checked;

            if(!name || !phone || !addr) return alert("Please fill Name, WhatsApp Number and Address");
            if(cart.length === 0) return alert("Cart is empty");
            if(!presc) return alert("Please check the 'I have a prescription' box to continue.");

            const items = cart.map(i => `• ${i.name} [x${i.qty}] - Rs.${i.price*i.qty}`).join('%0A');
            const total = cart.reduce((a,b)=>a+(b.price*b.qty), 0);
            
            const msg = `*NEW ORDER: ${siteSettings.brandName}*%0A%0A*Name:* ${name}%0A*Phone:* ${phone}%0A*Address:* ${addr}%0A%0A*Items:*%0A${items}%0A%0A*TOTAL: Rs.${total}*%0A%0A_Prescription Verified: YES_`;
            
            window.open(`https://wa.me/${siteSettings.waNumber}?text=${msg}`, '_blank');
        }

        function saveMedicine() {
            const name = document.getElementById('pName').value;
            const price = Number(document.getElementById('pPrice').value);
            if (!name || !price) return alert("Name and Price are required");
            const data = {
                name, price, category: document.getElementById('pCat').value || "General",
                img: document.getElementById('pImg').value, desc: document.getElementById('pDesc').value
            };
            if (currentEditId) db.ref('inventory/' + currentEditId).set(data);
            else db.ref('inventory').push(data);
            resetForm();
        }

        function editMedicine(id) {
            const p = inventory.find(i => i.id === id);
            currentEditId = id;
            document.getElementById('pName').value = p.name;
            document.getElementById('pPrice').value = p.price;
            document.getElementById('pCat').value = p.category;
            document.getElementById('pImg').value = p.img || "";
            document.getElementById('pDesc').value = p.desc || "";
            document.getElementById('formTitle').innerText = "Update Product";
            document.getElementById('saveBtn').innerText = "Save Changes";
            document.getElementById('cancelBtn').classList.remove('hidden');
        }

        function resetForm() {
            currentEditId = null;
            document.getElementById('formTitle').innerText = "Manage Product";
            document.getElementById('saveBtn').innerText = "Add Product";
            document.getElementById('cancelBtn').classList.add('hidden');
            ['pName', 'pPrice', 'pCat', 'pImg', 'pDesc'].forEach(id => document.getElementById(id).value = "");
        }
